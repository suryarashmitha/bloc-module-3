1. You and a friend have set a wager to see who can find the word "Albatross" in the dictionary the fastest. Write a program to allow you to win the bet.

A:

function dictionarySearch(array, word) {
  var low = array[0];
  var high = array[array.length - 1];

  while (low <= high) {
    var middle = (array.indexOf(low) + array.indexOf(high)) / 2;
    middle = Math.round(middle);

    if (array[middle] > word) {
      high = array[middle - 1];
    } else if (array[middle] < word) {
      low = array[middle + 1];
    } else {
      return array[middle];
    }
  }
  return 'That name is not in the list';
}

var dictionary = ['ability', 'abroad', 'account', 'aggressive', 'aim', 'air', 'albatross'];
console.log(dictionarySearch(dictionary, 'ability'));

2. You work at a pet store, and a child has asked you to net the only white-spotted goldfish from the fish tank. Write a program that will help you net the right fish.

A:

function findFish(tank, fishToFind) {
  for (var i = 0; i < tank.length; i++) {
    if (tank[i] === fishToFind) {
      return "You caught a " + fishToFind;
    }
  }
  return "There is no " + fishToFind + " in the tank";
}

var fishTank = ['gold fish', 'white fish', 'blue fish', 'red fish', 'white gold fish', 'silver fish'];

console.log(findFish(fishTank, 'white red gold fish'));
